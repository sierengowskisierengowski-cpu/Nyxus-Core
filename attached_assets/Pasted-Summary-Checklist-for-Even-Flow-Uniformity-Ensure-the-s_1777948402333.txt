Configuration Guide (Where to put the code)Hyprland Core SettingsFile: ~/.config/hypr/hyprland.confThis file controls the physical "tiled box" look and the frosted glass material.ini# --- Borders & Gaps ---
general {
    gaps_in = 10
    gaps_out = 20           # Large gaps allow the "Seattle Mist" to flow around windows
    border_size = 1         # Keep it thin like a sketch line
    col.active_border = rgba(d1d1d188) 
    col.inactive_border = rgba(31313144) 
    layout = dwindle
}

# --- Frosted Glass Physics ---
decoration {
    rounding = 15           # Softens the edges of your "sketch"
    active_opacity = 0.85
    inactive_opacity = 0.70

    blur {
        enabled = true
        size = 12           # Thicker glass
        passes = 4          # More passes = smoother frosting
        new_optimizations = true
        
        brightness = 1.4    # THE KEY: This creates the "Milk White" tint
        vibrancy = 0.2      # Low vibrancy keeps colors from leaking too much
        noise = 0.04        # Adds the physical grain of etched glass
    }
}
