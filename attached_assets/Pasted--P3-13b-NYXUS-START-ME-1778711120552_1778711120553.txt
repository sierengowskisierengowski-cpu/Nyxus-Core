;; P3.13b · NYXUS START MENU (Win11-grade launcher)
;; Bound to bottom-left "✕ NYXUS" brand button. STARTAPPS poller
;; produces pinned/recents/all/user. STARTQUERY filters `all`.
;; ═════════════════════════════════════════════════════════════════════

(defwidget startmenu_tile [name glyph exec]
  (button :class "startmenu-tile"
          :onclick "${exec} & disown ; eww close startmenu"
          :tooltip "${name}"
    (box :orientation "v" :space-evenly false :spacing 2 :halign "center"
      (label :class "startmenu-tile-glyph" :text "${glyph}")
      (label :class "startmenu-tile-label" :text "${name}" :limit-width 12))))

(defwidget startmenu_recent_row [name path exec]
  (button :class "startmenu-recent-row"
          :onclick "${exec} & disown ; eww close startmenu"
    (box :orientation "v" :space-evenly false :spacing 1 :halign "start" :hexpand true
      (label :class "startmenu-recent-name" :text "${name}" :halign "start" :limit-width 56)
      (label :class "startmenu-recent-path" :text "${path}" :halign "start" :limit-width 56))))

(defwidget startmenu_layout []
  (box :class "startmenu-root" :orientation "v" :space-evenly false :spacing 8
    ;; Search bar — typing updates STARTQUERY which the script re-reads.
    (box :orientation "h" :space-evenly false :spacing 8
      (label :class "startmenu-tile-glyph" :text "")
      (input :class "startmenu-search"
             :hexpand true
             :value STARTQUERY
             :onchange "eww update STARTQUERY={} ; ~/.config/eww/scripts/startmenu-apps.sh search {} >/dev/null &"))

    ;; Pinned grid (4 columns × 3 rows = 12 slots).
    (label :class "startmenu-section" :text "PINNED" :halign "start"
           :visible {STARTQUERY == ""})
    (box :class "startmenu-grid" :orientation "v" :space-evenly false :spacing 4
         :visible {STARTQUERY == ""}
      (for row in {STARTAPPS.pinned}
        (box :orientation "h" :space-evenly true :spacing 6
          (for app in {row}
            (startmenu_tile :name {app.name} :glyph {app.glyph} :exec {app.exec})))))

    ;; Recents — most-recent .desktop launches.
    (label :class "startmenu-section" :text "RECENT" :halign "start"
           :visible {STARTQUERY == "" && arraylength(STARTAPPS.recents) > 0})
    (box :orientation "v" :space-evenly false :spacing 2
         :visible {STARTQUERY == "" && arraylength(STARTAPPS.recents) > 0}
      (for r in {STARTAPPS.recents}
        (startmenu_recent_row :name {r.name} :path {r.path} :exec {r.exec})))

    ;; Search results — flat list when STARTQUERY != "".
    (label :class "startmenu-section" :text "RESULTS" :halign "start"
           :visible {STARTQUERY != ""})
    (scroll :vscroll true :hscroll false :height 320
            :visible {STARTQUERY != ""}
      (box :orientation "v" :space-evenly false :spacing 2
        (for a in {STARTAPPS.all}
          (startmenu_recent_row :name {a.name} :path {a.exec} :exec {a.exec}))))

    ;; Footer — user identity + power buttons.
    (box :class "startmenu-footer" :orientation "h" :space-evenly false :spacing 10
      (label :class "startmenu-user" :text "  ${STARTAPPS.user}@${STARTAPPS.host}"
             :hexpand true :halign "start")
      (button :class "startmenu-power-btn"
              :tooltip "Lock"
              :onclick "eww close startmenu ; hyprlock &" "")
      (button :class "startmenu-power-btn"
              :tooltip "Settings"
              :onclick "eww close startmenu ; nyxus-settings &" "")
      (button :class "startmenu-power-btn shutdown"
              :tooltip "Power"
              :onclick "eww close startmenu ; eww open --toggle powermenu" ""))))

  (startmenu_layout))
